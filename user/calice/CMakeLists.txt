list(INSERT CMAKE_MODULE_PATH 0 ${CMAKE_CURRENT_LIST_DIR}/cmake)
get_filename_component(EUDAQ_USERNAME ${CMAKE_CURRENT_LIST_DIR} NAME)
set(EUDAQ_MODULE ${EUDAQ_PREFIX}module_${EUDAQ_USERNAME})
option(USER_CALICE_BUILD "build user/calice" OFF)

if(NOT USER_CALICE_BUILD)
  message(STATUS "user/calice is NOT to be built (USER_CALICE_BUILD=OFF)")
  unset(USER_CALICE_BUILD_AHCAL CACHE)
  unset(USER_CALICE_BUILD_BIF CACHE)
  return()
else()
    message(STATUS "user/calice is to be built (USER_CALICE_BUILD=ON)")
endif()

add_subdirectory(module)
add_subdirectory(AHCAL)
add_subdirectory(BIF_AHCAL)
